function renderMeetingStatus() { const list = document.getElementById("meeting-list"); list.innerHTML = ""; meetingTypes.forEach(name => { const row = document.createElement("div"); row.className = "meeting-item"; // Meeting name label const meetingLabel = document.createElement("span"); meetingLabel.textContent = name; meetingLabel.style.width = "180px"; row.appendChild(meetingLabel); // Week 1 to Week 5 boxes for (let i = 1; i <= 5; i++) { const label = document.createElement("span"); label.textContent = `Week ${i}`; label.style.fontSize = "12px"; label.style.marginBottom = "4px"; const box = document.createElement("div"); box.className = "status-box"; box.title = `Click to toggle Week ${i} status`; box.onclick = () => { if (box.style.background === "gray") box.style.background = "green"; else if (box.style.background === "green") box.style.background = "red"; else box.style.background = "gray"; }; const wrapper = document.createElement("div"); wrapper.style.display = "flex"; wrapper.style.flexDirection = "column"; wrapper.style.alignItems = "center"; wrapper.style.marginRight = "12px"; wrapper.appendChild(label); wrapper.appendChild(box); row.appendChild(wrapper); } list.appendChild(row); }); }
